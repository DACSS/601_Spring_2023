  state_county_data <- read_xls('_data/StateCounty2012.xls', skip=4,

  state_county_data <- state_county_data %>%

  state_county_data <- state_county_data %>%

  state_county_data <- state_county_data %>%

  state_county_data <- state_county_data %>%
